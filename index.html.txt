        async function openPackAPI(userId, packId) {
            const pack = PACK_DATA[packId];
            if (!pack) {
                return { success: false, message: '–ù–∞–±–æ—Ä –Ω–µ –Ω–∞–π–¥–µ–Ω.' };
            }
            if (MOCK_BALANCE < pack.cost) {
                return { success: false, message: '–ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –ó–≤–µ–∑–¥!' };
            }
            if (pack.stock !== Infinity && pack.stock <= 0) {
                return { success: false, message: '–ù–∞–±–æ—Ä –∑–∞–∫–æ–Ω—á–∏–ª—Å—è!' };
            }

            // 1. –°–ø–∏—Å–∞–Ω–∏–µ —Å—Ç–æ–∏–º–æ—Å—Ç–∏
            MOCK_BALANCE -= pack.cost;

            // 2. –û–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –ø–æ–ª—É—á–µ–Ω–Ω–æ–≥–æ –ø—Ä–µ–¥–º–µ—Ç–∞ (–ø—Ä–æ—Å—Ç–∞—è –ª–æ–≥–∏–∫–∞ —à–∞–Ω—Å–æ–≤)
            const randomValue = Math.random() * 100;
            let cumulativeChance = 0;
            let receivedItem = null;

            for (const reward of pack.rewards) {
                cumulativeChance += reward.chance;
                if (randomValue <= cumulativeChance) {
                    receivedItem = reward;
                    break;
                }
            }

            // 3. –û–±—Ä–∞–±–æ—Ç–∫–∞ –ø–æ–ª—É—á–µ–Ω–∏—è
            if (receivedItem.type === 'Stars') {
                MOCK_BALANCE += receivedItem.amount;
            } else {
                // –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –≤ –∏–Ω–≤–µ–Ω—Ç–∞—Ä—å (–∏–ª–∏ —É–≤–µ–ª–∏—á–µ–Ω–∏–µ —Å—á–µ—Ç—á–∏–∫–∞)
                const existing = MOCK_INVENTORY.find(item => item.name === receivedItem.name);
                if (existing) {
                    existing.count += 1;
                } else {
                    MOCK_INVENTORY.push({ ...receivedItem, id: Date.now(), count: 1 });
                }
            }

            // 4. –£–≤–µ–¥–æ–º–ª–µ–Ω–∏–µ –ê–¥–º–∏–Ω–∏—Å—Ç—Ä–∞—Ç–æ—Ä–∞ (–ö–ª—é—á–µ–≤–æ–µ —É—Å–ª–æ–≤–∏–µ)
            if (receivedItem.type === 'NFT' || receivedItem.type === 'Gift') {
                console.warn(`[ADMIN NOTIFICATION] –ü–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å ${userId} –ø–æ–ª—É—á–∏–ª –†–ï–î–ö–ò–ô –ü–†–ò–ó: ${receivedItem.name}. –¢—Ä–µ–±—É–µ—Ç—Å—è —Ä—É—á–Ω–∞—è –≤—ã–¥–∞—á–∞.`);
                if (window.Telegram && window.Telegram.WebApp) {
                    Telegram.WebApp.showAlert(`–í–Ω–∏–º–∞–Ω–∏–µ! –í—ã –ø–æ–ª—É—á–∏–ª–∏ —Ä–µ–¥–∫–∏–π –ø—Ä–∏–∑ (${receivedItem.name})! –°–≤—è–∂–∏—Ç–µ—Å—å —Å –∞–¥–º–∏–Ω–∏—Å—Ç—Ä–∞—Ç–æ—Ä–æ–º –¥–ª—è –≤—ã–¥–∞—á–∏.`);
                }
            }

            // 5. –£–º–µ–Ω—å—à–µ–Ω–∏–µ –ª–∏–º–∏—Ç–∞ (–µ—Å–ª–∏ –ª–∏–º–∏—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π)
            if (pack.stock !== Infinity) {
                pack.stock -= 1;
            }

            return { success: true, item: receivedItem };
        }
